create or replace FUNCTION PROJECT_VALIDATE_HEIGHT 
(
  HEIGHT IN VARCHAR2 
) RETURN BOOLEAN AS 
BEGIN
  IF REGEXP_LIKE (HEIGHT, '^\d+['',"]\s?(\d{1,2}")?$') THEN
  RETURN TRUE;
  ELSE 
  RETURN FALSE;
  END IF;
END PROJECT_VALIDATE_HEIGHT;

create or replace FUNCTION PROJECT_VALIDATE_PERCENTAGE 
(
  PERCENTAGE_FIELD IN VARCHAR2 
) RETURN BOOLEAN AS 
BEGIN
  IF REGEXP_LIKE (PERCENTAGE_FIELD, '\\d+(?:\\.\\d+)?%') THEN
  RETURN TRUE;
  ELSE 
  RETURN FALSE;
  END IF;
END PROJECT_VALIDATE_PERCENTAGE;

create or replace FUNCTION PROJECT_VALIDATE_TIME 
(
  T_IME IN VARCHAR2 
) RETURN BOOLEAN AS 
BEGIN
  IF REGEXP_LIKE (T_IME, '^\d+:[0-5][0-9]?$') THEN
  RETURN TRUE;
  ELSE 
  RETURN FALSE;
  END IF;
END PROJECT_VALIDATE_TIME;
